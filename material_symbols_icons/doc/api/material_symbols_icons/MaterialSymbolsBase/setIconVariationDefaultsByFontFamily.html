static void setIconVariationDefaultsByFontFamily(
    String fontFamily, IconVariationDefaults? variations) {
  if (variations == null) {
    globalIconVariationDefaults.remove(fontFamily);
  } else {
    globalIconVariationDefaults[fontFamily] = variations;
  }
}
